#defaultView:ImageGrid
SELECT DISTINCT ?peinture ?peintureLabel ?image ?natureLabel ?mouvementLabel ?article
(MD5(CONCAT(str(?peinture),str(RAND()))) as ?random)
WHERE {
  ?peinture wdt:P135 ?mouvement;
             wdt:P31 wd:Q3305213;
             wdt:P31 ?nature;
             wdt:P18 ?image.

  ?article schema:about ?peinture;
    schema:inLanguage "fr";
    schema:isPartOf <https://fr.wikipedia.org/>.

  SERVICE wikibase:label { bd:serviceParam wikibase:language "fr". }
  FILTER ( ?mouvement in (wd:Q1404472, wd:Q5668507, wd:Q122353293, wd:Q42934,  wd:Q40415,  wd:Q808561, wd:Q2642826, wd:Q765606, wd:Q577147, wd:Q947129, wd:Q3088366, wd:Q19953535))
}
ORDER BY ?random
